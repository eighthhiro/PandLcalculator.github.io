Profit and Loss Calculator




def calculate_profit_loss(event): # Get input values income = Element("income").element.value expenses = Element("expenses").element.value # Validate input and calculate profit or loss if income and expenses: # Ensure both inputs are filled income = float(income) expenses = float(expenses) profit_loss = income - expenses if profit_loss > 0: result_text = f"Profit: ${profit_loss:.2f}" elif profit_loss < 0: result_text = f"Loss: ${-profit_loss:.2f}" else: result_text = "Break-even: No profit, no loss." else: result_text = "Please enter valid values for both income and expenses." # Display the result Element("result").element.innerHTML = result_text # Bind the function to the button click event Element("calculate-btn").element.addEventListener("click", calculate_profit_loss)
